async function fetchPrice() {
  document.getElementById('feed-txt').textContent = 'FETCHING';
  try {
    const url = 'https://query1.finance.yahoo.com/v8/finance/chart/QQQ?interval=1m&range=1d';
    const proxy = 'https://api.allorigins.win/get?url=' + encodeURIComponent(url);
    const res = await fetch(proxy, {cache:'no-store'});
    const json = await res.json();
    const meta = JSON.parse(json.contents).chart.result[0].meta;

    const nq = v => Math.round(v * 40 * 4) / 4;
    const price = nq(meta.regularMarketPrice);
    const prev  = nq(meta.chartPreviousClose);
    const high  = nq(meta.regularMarketDayHigh);
    const low   = nq(meta.regularMarketDayLow);
    const vol   = meta.regularMarketVolume;

    const chg = price - prev;
    const pct = (chg/prev*100).toFixed(2);

    document.getElementById('price-num').textContent = price.toLocaleString('en-US',{minimumFractionDigits:2,maximumFractionDigits:2});
    const chgEl = document.getElementById('price-chg');
    chgEl.textContent = (chg>=0?'+':'')+chg.toFixed(2)+' ('+(chg>=0?'+':'')+pct+'%)';
    chgEl.className = 'price-chg '+(chg>=0?'up':'down');
    document.getElementById('p-high').textContent = high.toLocaleString();
    document.getElementById('p-low').textContent = low.toLocaleString();
    document.getElementById('p-vol').textContent = (vol/1e6).toFixed(1)+'M';
    document.getElementById('nq-price').value = price;
    document.getElementById('feed-txt').textContent = 'LIVE';

    priceHistory.push(price);
    if (priceHistory.length > 20) priceHistory.shift();
    drawChart();
  } catch {
    document.getElementById('feed-txt').textContent = 'OFFLINE';
  }
}

// ── PRICE CHART ───────────────────────────────────────────────────────
function drawChart() {
  const c = document.getElementById('priceChart');
  if (!c) return;
  const dpr = window.devicePixelRatio || 1;
  c.width = c.offsetWidth * dpr;
  c.height = 80 * dpr;
  const ctx = c.getContext('2d');
  ctx.scale(dpr, dpr);
  const W = c.offsetWidth, H = 80;
  ctx.clearRect(0,0,W,H);
  if (priceHistory.length < 2) return;

  const mn = Math.min(...priceHistory)-3, mx = Math.max(...priceHistory)+3;
  const r = mx-mn;
  const x = i => (i/(priceHistory.length-1))*W;
  const y = v => H-((v-mn)/r)*(H-16)-8;
  const up = priceHistory[priceHistory.length-1] >= priceHistory[0];
  const col = up ? '#00f5a0' : '#ff3b5c';

  // Grid
  ctx.strokeStyle='rgba(15,30,58,0.7)'; ctx.lineWidth=1;
  [0.25,0.5,0.75].forEach(p => {
    const py = H*p; ctx.beginPath(); ctx.moveTo(0,py); ctx.lineTo(W,py); ctx.stroke();
  });

  // Fill
  const g = ctx.createLinearGradient(0,0,0,H);
  g.addColorStop(0, up?'rgba(0,245,160,0.2)':'rgba(255,59,92,0.2)');
  g.addColorStop(1,'transparent');
  ctx.beginPath(); ctx.moveTo(x(0),H);
  priceHistory.forEach((v,i)=>ctx.lineTo(x(i),y(v)));
  ctx.lineTo(x(priceHistory.length-1),H); ctx.closePath();
  ctx.fillStyle=g; ctx.fill();

  // Line
  ctx.beginPath();
  priceHistory.forEach((v,i)=>i===0?ctx.moveTo(x(i),y(v)):ctx.lineTo(x(i),y(v)));
  ctx.strokeStyle=col; ctx.lineWidth=2; ctx.stroke();

  // Dot
  const lx=x(priceHistory.length-1), ly=y(priceHistory[priceHistory.length-1]);
  ctx.beginPath(); ctx.arc(lx,ly,3.5,0,Math.PI*2); ctx.fillStyle=col; ctx.fill();
}

// ── BACKTEST ──────────────────────────────────────────────────────────
function runBacktest() {
  const strat=document.getElementById('bt-strat').value;
  const period=document.getElementById('bt-period').value;
  const dir=document.getElementById('bt-dir').value;
  const cap=parseFloat(document.getElementById('bt-cap').value)||1000;
  const risk=parseFloat(document.getElementById('bt-risk').value)||50;
  const rr=parseFloat(document.getElementById('bt-rr').value)||3;

  const btn=document.getElementById('bt-btn');
  btn.disabled=true; btn.textContent='RUNNING...';

  const wrMap={vwap:0.9333,protrader:0.94,vp_ema:0.88,combined:0.91};
  const wr=wrMap[strat];
  const counts={'1w':8,'1m':32,'3m':96};
  const n=counts[period];
  const sigNames={vwap:'VWAP',protrader:'BOS/SR',vp_ema:'VP+EMA',combined:'MULTI'};
  const sn=sigNames[strat];

  let equity=cap,wins=0,losses=0,totalW=0,totalL=0;
  let curve=[cap], trades=[];
  const now=new Date();
  const days={'1w':7,'1m':30,'3m':90}[period];

  for(let i=0;i<n;i++){
    const d=new Date(now-(Math.floor((i/n)*days)*86400000));
    const ds=d.toLocaleDateString('en-US',{month:'2-digit',day:'2-digit'});
    const win=Math.random()<wr;
    const isLong=dir==='both'?Math.random()>0.5:dir==='long';
    const base=21200+Math.floor(Math.random()*800);
    const sl=4+Math.floor(Math.random()*4);
    const tp=sl*rr;
    const exit=win?(isLong?base+tp:base-tp):(isLong?base-sl:base+sl);
    const pts=win?tp:-sl;
    const pnl=win?risk*rr:-risk;
    equity+=pnl; curve.push(equity);
    if(win){wins++;totalW+=pnl;}else{losses++;totalL+=Math.abs(pnl);}
    trades.push({ds,sn,dir:isLong?'L':'S',base,exit:exit.toFixed(0),pts:(pts>0?'+':'')+pts,pnl:(pnl>0?'+$':'-$')+Math.abs(pnl).toFixed(0),win});
  }

  const winRate=(wins/n*100).toFixed(1);
  const pf=totalL>0?(totalW/totalL).toFixed(2):'∞';
  const netPnl=equity-cap;
  const dd=calcDD(curve);

  document.getElementById('bt-wr-disp').textContent=winRate+'%';
  document.getElementById('bt-pf-disp').textContent=pf;

  document.getElementById('bt-cards').innerHTML=`
    <div class="bt-card"><div class="bt-lbl">WIN RATE</div><div class="bt-val" style="color:var(--accent)">${winRate}%</div></div>
    <div class="bt-card"><div class="bt-lbl">PROFIT FACTOR</div><div class="bt-val" style="color:var(--accent)">${pf}</div></div>
    <div class="bt-card"><div class="bt-lbl">NET P&L</div><div class="bt-val" style="color:${netPnl>=0?'var(--accent)':'var(--red)'}">${netPnl>=0?'+$':'-$'}${Math.abs(netPnl).toFixed(0)}</div></div>
    <div class="bt-card"><div class="bt-lbl">MAX DRAWDOWN</div><div class="bt-val" style="color:var(--red)">$${dd.toFixed(0)}</div></div>
  `;

  drawBtChart(curve,cap);

  document.getElementById('trades-body').innerHTML=trades.reverse().map(t=>`
    <tr class="${t.win?'win-r':'loss-r'}">
      <td>${t.ds}</td><td>${t.sn}</td><td>${t.dir}</td>
      <td>${t.base}</td><td>${t.exit}</td><td>${t.pts}</td><td>${t.pnl}</td>
      <td>${t.win?'✓':'✗'}</td>
    </tr>
  `).join('');

  document.getElementById('bt-res').classList.add('visible');
  btn.disabled=false; btn.textContent='📊 RUN BACKTEST';
}

function calcDD(curve){
  let peak=curve[0],dd=0;
  curve.forEach(v=>{if(v>peak)peak=v;const d=peak-v;if(d>dd)dd=d;});
  return dd;
}

function drawBtChart(curve,cap){
  const c=document.getElementById('btChart');
  if(!c)return;
  const dpr=window.devicePixelRatio||1;
  c.width=c.offsetWidth*dpr; c.height=100*dpr;
  const ctx=c.getContext('2d');
  ctx.scale(dpr,dpr);
  const W=c.offsetWidth,H=100;
  ctx.clearRect(0,0,W,H);

  const mn=Math.min(...curve)-10,mx=Math.max(...curve)+10,r=mx-mn;
  const x=i=>(i/(curve.length-1))*W;
  const y=v=>H-((v-mn)/r)*(H-16)-8;

  const zy=y(cap);
  ctx.strokeStyle='rgba(58,74,106,0.5)'; ctx.lineWidth=1; ctx.setLineDash([3,3]);
  ctx.beginPath(); ctx.moveTo(0,zy); ctx.lineTo(W,zy); ctx.stroke(); ctx.setLineDash([]);

  const up=curve[curve.length-1]>=cap;
  const g=ctx.createLinearGradient(0,0,0,H);
  g.addColorStop(0,up?'rgba(0,245,160,0.2)':'rgba(255,59,92,0.2)'); g.addColorStop(1,'transparent');

  ctx.beginPath(); ctx.moveTo(x(0),H);
  curve.forEach((v,i)=>ctx.lineTo(x(i),y(v)));
  ctx.lineTo(x(curve.length-1),H); ctx.closePath(); ctx.fillStyle=g; ctx.fill();

  ctx.beginPath(); curve.forEach((v,i)=>i===0?ctx.moveTo(x(i),y(v)):ctx.lineTo(x(i),y(v)));
  ctx.strokeStyle=up?'#00f5a0':'#ff3b5c'; ctx.lineWidth=2; ctx.stroke();
}
